<script>
  // =====================================================
  //  FIRE CANVAS — 불꽃 + 불티(ember) + 연기 파티클 시스템
  // =====================================================
  (function(){
    var reduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
    if(reduced) return;

    var canvas = document.getElementById('fire-canvas');
    var ctx = canvas.getContext('2d');
    var W, H;

    function resize(){
      W = canvas.width = window.innerWidth;
      H = canvas.height = window.innerHeight;
    }
    resize();
    window.addEventListener('resize', resize);

    var rand = function(a,b){ return a + Math.random()*(b-a); };

    // ── 파티클 공통 풀 ──────────────────────────────────
    var particles = [];

    // ── 1. 불꽃 파티클 (Flame) ───────────────────────────
    //    하단 3개 발생원(좌/중/우)에서 위로 올라가며
    //    주황→황→투명, 길쭉한 타원형, 흔들림
    function Flame(srcX){
      this.reset(srcX);
    }
    Flame.prototype.reset = function(srcX){
      this.type  = 'flame';
      this.srcX  = srcX !== undefined ? srcX : rand(W*0.1, W*0.9);
      this.x     = this.srcX + rand(-40, 40);
      this.y     = H + rand(0, 30);
      this.vx    = rand(-0.6, 0.6);
      this.vy    = rand(-3.5, -1.8);
      this.life  = 1.0;
      this.decay = rand(0.008, 0.018);
      this.r     = rand(18, 52);       // 반지름
      this.scaleY= rand(1.6, 2.8);     // 위로 늘린 비율
      this.wobble= rand(0, Math.PI*2); // 흔들림 위상
      this.wobbleSpeed = rand(0.04, 0.10);
    };
    Flame.prototype.update = function(){
      this.wobble += this.wobbleSpeed;
      this.x  += this.vx + Math.sin(this.wobble) * 0.7;
      this.y  += this.vy;
      this.vy *= 0.995;                // 위로 갈수록 느려짐
      this.r  *= 0.997;               // 크기 서서히 줄어듦
      this.life -= this.decay;
    };
    Flame.prototype.draw = function(){
      if(this.life <= 0) return;
      ctx.save();
      ctx.translate(this.x, this.y);
      ctx.scale(1, this.scaleY);
      var g = ctx.createRadialGradient(0,0, 0, 0,0, this.r);
      // 코어: 밝은 황백
      g.addColorStop(0,   'rgba(255,230,140,'+(this.life*0.85)+')');
      // 중간: 주황
      g.addColorStop(0.35,'rgba(255,120,20,'+  (this.life*0.70)+')');
      // 외곽: 심홍
      g.addColorStop(0.7, 'rgba(200,40,0,'+    (this.life*0.45)+')');
      // 끝: 투명
      g.addColorStop(1,   'rgba(80,10,0,0)');
      ctx.globalCompositeOperation = 'lighter';
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(0,0, this.r, 0, Math.PI*2);
      ctx.fill();
      ctx.restore();
    };

    // ── 2. 불티 파티클 (Ember / Spark) ──────────────────
    //    작고 밝은 점, 포물선 궤적으로 튀어오름
    function Ember(){
      this.reset();
    }
    Ember.prototype.reset = function(){
      this.type  = 'ember';
      this.x     = rand(W*0.05, W*0.95);
      this.y     = H + rand(0, 60);
      this.vx    = rand(-2.2, 2.2);
      this.vy    = rand(-7, -2.5);
      this.gravity = rand(0.04, 0.09);
      this.life  = 1.0;
      this.decay = rand(0.006, 0.016);
      this.r     = rand(1.2, 3.2);
      this.trail = [];
    };
    Ember.prototype.update = function(){
      // 잔상 기록
      this.trail.push({x: this.x, y: this.y});
      if(this.trail.length > 6) this.trail.shift();

      this.x  += this.vx;
      this.y  += this.vy;
      this.vy += this.gravity; // 중력
      this.vx *= 0.99;
      this.life -= this.decay;
    };
    Ember.prototype.draw = function(){
      if(this.life <= 0) return;
      // 잔상(꼬리)
      for(var i=0;i<this.trail.length;i++){
        var t = i/this.trail.length;
        ctx.save();
        ctx.globalCompositeOperation = 'lighter';
        ctx.beginPath();
        ctx.arc(this.trail[i].x, this.trail[i].y, this.r*t*0.7, 0, Math.PI*2);
        ctx.fillStyle = 'rgba(255,160,30,'+(this.life*t*0.4)+')';
        ctx.fill();
        ctx.restore();
      }
      // 본체
      ctx.save();
      ctx.globalCompositeOperation = 'lighter';
      var g = ctx.createRadialGradient(this.x,this.y,0, this.x,this.y, this.r*2.5);
      g.addColorStop(0,   'rgba(255,255,200,'+(this.life)+')');
      g.addColorStop(0.4, 'rgba(255,140,20,'+(this.life*0.8)+')');
      g.addColorStop(1,   'rgba(255,60,0,0)');
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.r*2.5, 0, Math.PI*2);
      ctx.fill();
      ctx.restore();
    };

    // ── 3. 연기 파티클 (Smoke) ───────────────────────────
    //    불꽃 위쪽에서 생성, 느리고 크게 퍼짐, 어둡고 투명
    function Smoke(){
      this.reset();
    }
    Smoke.prototype.reset = function(){
      this.type  = 'smoke';
      this.x     = rand(W*0.15, W*0.85);
      this.y     = rand(H*0.55, H*0.75);  // 중하단에서 생성
      this.vx    = rand(-0.3, 0.3);
      this.vy    = rand(-0.6, -0.2);
      this.life  = rand(0.15, 0.35);
      this.decay = rand(0.0015, 0.004);
      this.r     = rand(30, 80);
      this.grow  = rand(0.3, 0.8);
    };
    Smoke.prototype.update = function(){
      this.x  += this.vx;
      this.y  += this.vy;
      this.r  += this.grow;
      this.vx += rand(-0.02, 0.02);
      this.life -= this.decay;
    };
    Smoke.prototype.draw = function(){
      if(this.life <= 0) return;
      ctx.save();
      var g = ctx.createRadialGradient(this.x,this.y,0, this.x,this.y,this.r);
      g.addColorStop(0,   'rgba(60,50,40,'+(this.life*0.5)+')');
      g.addColorStop(0.5, 'rgba(40,32,26,'+(this.life*0.3)+')');
      g.addColorStop(1,   'rgba(20,15,10,0)');
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.r, 0, Math.PI*2);
      ctx.fill();
      ctx.restore();
    };

    // ── 발생원 정의 (하단 3곳 + 무작위) ────────────────
    var flameSources = [
      W * 0.20,
      W * 0.50,
      W * 0.80,
    ];

    // 초기 풀 채우기
    for(var i=0;i<60;i++){
      var f = new Flame(flameSources[i%3]);
      f.life = rand(0, 1);            // stagger
      particles.push(f);
    }
    for(var i=0;i<80;i++){
      var e = new Ember();
      e.life = rand(0, 1);
      particles.push(e);
    }
    for(var i=0;i<35;i++){
      var s = new Smoke();
      s.life = rand(0, s.life);
      particles.push(s);
    }

    // 발생원 X는 viewport resize에 따라 갱신
    function updateSources(){
      flameSources[0] = W * 0.20;
      flameSources[1] = W * 0.50;
      flameSources[2] = W * 0.80;
    }
    window.addEventListener('resize', updateSources);

    // ── 메인 루프 ───────────────────────────────────────
    function tick(){
      ctx.clearRect(0, 0, W, H);

      for(var i=0;i<particles.length;i++){
        var p = particles[i];
        p.update();
        p.draw();

        if(p.life <= 0){
          // 재활용
          if(p.type === 'flame'){
            var src = flameSources[Math.floor(rand(0,3))];
            p.reset(src);
          } else {
            p.reset();
          }
        }
      }

      requestAnimationFrame(tick);
    }
    tick();
  })();

  // 유튜브 방식 재생: 클릭 시 재생, 동시에 하나만
  (function(){
    var players = document.querySelectorAll('[data-player]');
    var allVideos = [];

    players.forEach(function(p){
      var video = p.querySelector('video');
      var overlay = p.querySelector('.play-overlay');
      allVideos.push(video);

      function start(){
        allVideos.forEach(function(v){
          if(v !== video){ v.pause(); }
        });
        document.querySelectorAll('.play-overlay').forEach(function(o){
          if(o !== overlay){ o.classList.remove('hidden'); }
        });
        video.controls = true;
        overlay.classList.add('hidden');
        video.play();
      }

      overlay.addEventListener('click', start);
      overlay.addEventListener('keydown', function(e){
        if(e.key === 'Enter' || e.key === ' '){ e.preventDefault(); start(); }
      });

      video.addEventListener('pause', function(){
        if(video.currentTime === 0 || video.ended){
          overlay.classList.remove('hidden');
          video.controls = false;
        }
      });
      video.addEventListener('ended', function(){
        overlay.classList.remove('hidden');
        video.controls = false;
      });
    });
  })();
</script>
